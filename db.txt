-- Drop RLS policies first
DO $$
DECLARE
    pol RECORD;
BEGIN
    FOR pol IN
        SELECT tablename, policyname
        FROM pg_policies
        WHERE schemaname = 'public'
          AND tablename IN ('trades', 'bot_state', 'signals', 'profiles', 'mt5_credentials', 'support_tickets', 'system_logs', 'news_state', 'legal_acceptances', 'user_settings')
    LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON public.%I', pol.policyname, pol.tablename);
    END LOOP;
END $$;

-- Drop triggers
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
DROP FUNCTION IF EXISTS handle_new_user();

-- Drop tables (order matters due to FK dependencies)
DROP TABLE IF EXISTS public.legal_acceptances CASCADE;
DROP TABLE IF EXISTS public.support_tickets CASCADE;
DROP TABLE IF EXISTS public.system_logs CASCADE;
DROP TABLE IF EXISTS public.news_state CASCADE;

DROP TABLE IF EXISTS public.bot_state CASCADE;
DROP TABLE IF EXISTS public.trades CASCADE;
DROP TABLE IF EXISTS public.profiles CASCADE;

-- ============================================================
-- PART 2: CREATE TABLES
-- ============================================================

-- ── 1. profiles ─────────────────────────────────────────────
-- User profiles, linked to Supabase Auth. Created automatically on signup.
CREATE TABLE public.profiles (
    id                  UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    risk_percent        FLOAT DEFAULT 5.0,
    be_policy           TEXT DEFAULT 'auto',
    dry_run             BOOLEAN DEFAULT FALSE,
    auto_compounding    BOOLEAN DEFAULT FALSE,
    display_name        VARCHAR(100) DEFAULT 'Trader',
    notifications       JSONB DEFAULT '{"trade_execution": true, "trade_closed": true, "daily_summary": true, "loss_cooldown": true, "maintenance": true, "email_trade": false}'::jsonb,
    bot_active          BOOLEAN DEFAULT FALSE,
    expo_push_token     TEXT,
    broker_verified     BOOLEAN DEFAULT FALSE,
    broker_name         TEXT,
    created_at          TIMESTAMPTZ DEFAULT NOW(),
    updated_at          TIMESTAMPTZ DEFAULT NOW()
);

-- ── 2. trades ───────────────────────────────────────────────
-- All trade records. One row per MT5 ticket.
CREATE TABLE public.trades (
    ticket          BIGINT PRIMARY KEY,
    user_id         UUID REFERENCES public.profiles(id) ON DELETE SET NULL,
    pair            TEXT NOT NULL,
    direction       TEXT NOT NULL,
    lots            REAL NOT NULL,
    entry_price     REAL NOT NULL,
    sl_price        REAL,
    tp_price        REAL,
    close_price     REAL,
    pnl             REAL,
    status          TEXT NOT NULL DEFAULT 'OPEN',
    opened_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    closed_at       TIMESTAMPTZ,
    confidence      INTEGER DEFAULT 0,
    sectors_json    TEXT
);

-- ── 3. bot_state ────────────────────────────────────────────
-- Key-value store for bot settings, cooldowns, etc.
CREATE TABLE public.bot_state (
    key             TEXT PRIMARY KEY,
    value           TEXT NOT NULL,
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ── 4. signals ──────────────────────────────────────────────
-- Log of every signal the bot generates (even if not traded).
CREATE TABLE public.signals (
    id              BIGSERIAL PRIMARY KEY,
    pair            TEXT NOT NULL,
    direction       TEXT NOT NULL,
    confidence      INTEGER DEFAULT 0,
    alignment       TEXT,
    sectors_json    TEXT,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ── 5. mt5_credentials ──────────────────────────────────────
-- MT5 login credentials per user. Supports multiple accounts per user.
CREATE TABLE public.mt5_credentials (
    id                          BIGSERIAL PRIMARY KEY,
    user_id                     UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
    login                       VARCHAR(50) NOT NULL,
    password                    VARCHAR(500) NOT NULL,
    server                      VARCHAR(100) NOT NULL,
    account_name                VARCHAR(100),
    connected                   BOOLEAN DEFAULT FALSE,
    automated_trading_enabled   BOOLEAN DEFAULT FALSE,
    last_error                  TEXT,
    last_connected_at           TIMESTAMPTZ,
    updated_at                  TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, login, server)
);

-- ── 6. support_tickets ──────────────────────────────────────
-- User support requests.
CREATE TABLE public.support_tickets (
    id              BIGSERIAL PRIMARY KEY,
    user_id         UUID REFERENCES public.profiles(id) ON DELETE SET NULL,
    email           TEXT,
    subject         TEXT,
    message         TEXT,
    status          TEXT DEFAULT 'open',
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ── 6b. user_settings ──────────────────────────────────────
-- Per-user configuration overrides (max_daily_trades, etc.)
CREATE TABLE public.user_settings (
    user_id             UUID PRIMARY KEY REFERENCES public.profiles(id) ON DELETE CASCADE,
    max_daily_trades    INTEGER DEFAULT 5,
    risk_percent        FLOAT DEFAULT 5.0,
    trading_mode        TEXT DEFAULT 'short',
    updated_at          TIMESTAMPTZ DEFAULT NOW()
);

-- ── 7. system_logs ──────────────────────────────────────────
-- Admin action & error logging.
CREATE TABLE public.system_logs (
    id              BIGSERIAL PRIMARY KEY,
    user_id         TEXT,
    action          TEXT,
    detail          TEXT,
    level           TEXT DEFAULT 'INFO',
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ── 8. news_state ───────────────────────────────────────────
-- News monitoring phase state (used by Telegram admin bot).
CREATE TABLE public.news_state (
    id              BIGSERIAL PRIMARY KEY,
    phase           TEXT,
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ── 9. legal_acceptances ────────────────────────────────────
-- Records when a user views/accepts Terms & Risk Disclosure.
CREATE TABLE public.legal_acceptances (
    id                  BIGSERIAL PRIMARY KEY,
    user_id             UUID REFERENCES public.profiles(id) ON DELETE SET NULL,
    terms_version       TEXT DEFAULT '2.0',
    risk_version        TEXT DEFAULT '2.0',
    ip_address          TEXT,
    device_fingerprint  TEXT,
    scroll_verified     BOOLEAN DEFAULT FALSE,
    created_at          TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- PART 5: AUTO-CREATE PROFILE ON SIGNUP
-- ============================================================
-- When a new user signs up via Supabase Auth, this trigger
-- automatically creates a row in the profiles table.
-- broker_verified is FALSE by default — users must verify
-- through the broker-connect screen to enable trading.

CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
    INSERT INTO public.profiles (id)
    VALUES (NEW.id);
    RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW
    EXECUTE FUNCTION public.handle_new_user();

    ('risk_percent', '5.0')
ON CONFLICT (key) DO NOTHING;
